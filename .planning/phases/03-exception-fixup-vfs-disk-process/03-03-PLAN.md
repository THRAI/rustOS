use core::sync::atomic::{AtomicI32, AtomicU8, Ordering};
use spin::Mutex;

use crate::mm::vm::VmMap;
use crate::mm::pmap::Pmap;

/// Process states.
#[repr(u8)]
pub enum TaskState {
    Running = 0,
    Zombie = 1,
}

pub struct Task {
    pub pid: u32,
    pub parent: Mutex<Weak<Task>>,
    pub children: Mutex<Vec<Arc<Task>>>,
    pub vm_map: Mutex<VmMap>,
    pub pmap: Mutex<Pmap>,
    /// Atomic state for lock-free check in wait4.
    pub state: AtomicU8,
    /// Exit status, written with Release in sys_exit.

    /// Waker for parent's wait4 future.

}

Key design points per CONTEXT.md:
- `parent` is `Weak<Task>` — no circular Arc
- `state` and `exit_status` are atomic for lock-free reads in wait4 hot path
- Memory ordering: `exit_status.store(Release)` + `state.store(ZOMBIE, Release)` in exit; `state.load(Acquire)` in wait4
- `parent_waker` stores the Waker from WaitChildFuture for lost-wakeup prevention

**1. fork implementation** (`kernel/src/proc/fork.rs`):

```rust
pub fn fork(parent: &Arc<Task>) -> Result<Arc<Task>, Errno> { ... }
```

fork() does:
1. Allocate new PID via `alloc_pid()`
2. Clone the parent's VmMap via `vm_map.fork()` — this already exists and inserts shadow objects at chain head for COW
3. Create a new Pmap via `pmap_create()` — child gets fresh page tables
4. Walk the parent's VmMap and call `pmap_enter` for each resident page in the child's new Pmap, marking all writable pages as read-only (COW). Also downgrade parent's PTEs to read-only for shared pages.
5. Create child Task with `Weak::clone` of parent Arc as parent link
6. Add child to parent's children list
7. Return `Arc<Task>` for the child

Key: The VmMap fork (step 2) handles the VmObject shadow chain insertion. The Pmap work (steps 3-4) handles the hardware page table COW setup. Both parent and child PTEs must be read-only for writable VMAs after fork — write faults trigger COW in the existing sync_fault_handler.

**2. sys_exit** (`kernel/src/proc/exit_wait.rs`):

Per CONTEXT.md two-phase death:
```rust
pub async fn sys_exit(task: &Arc<Task>, code: i32) -> SyscallResult {
    // 1. Store exit status with Release ordering
    task.exit_status.store(code, Ordering::Release);

    // 2. Transition to ZOMBIE with Release ordering
    task.state.store(TaskState::Zombie as u8, Ordering::Release);

    // 3. Reparent living children to init (pid 1)
    task.reparent_children_to(&get_init_task());

    // 4. Wake parent's wait4 Waker (if any)
    if let Some(waker) = task.parent_waker.lock().take() {
        waker.wake();
    }
    // Actually: the waker is on the PARENT, not on self.
    // Correction: parent has a WaitChildFuture whose Waker is stored
    // somewhere the child can reach. Store it on the CHILD's parent_waker
    // field — the parent's WaitChildFuture stores its Waker on each child
    // it's waiting for. OR: store a single Waker on the parent Task that
    // any child can wake.
    //
    // Simpler design: parent Task has a `child_exit_waker: Mutex<Option<Waker>>`
    // field. wait4 stores its Waker there. sys_exit on any child wakes it.
    // This means: add `child_exit_waker` to Task struct.

    if let Some(parent) = task.parent.lock().upgrade() {
        if let Some(waker) = parent.child_exit_waker.lock().take() {
            waker.wake();
        }
    }

    // 5. Return Terminated — executor drops the future
    SyscallResult::Terminated
}
```

**3. wait4** (`kernel/src/proc/exit_wait.rs`):

Implement `WaitChildFuture` per CONTEXT.md lost-wakeup prevention:

```rust
pub struct WaitChildFuture {
    parent: Arc<Task>,
    target_pid: Option<u32>, // None = any child, Some(pid) = specific
}

impl Future for WaitChildFuture {
    type Output = (u32, i32); // (pid, exit_status)

    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
        let parent = &self.parent;

        // CRITICAL: Register Waker FIRST, then scan.
        // This prevents the gap between "check" and "suspend".
        *parent.child_exit_waker.lock() = Some(cx.waker().clone());

        // Scan children for ZOMBIE
        let children = parent.children.lock();
        for child in children.iter() {
            if child.state.load(Ordering::Acquire) == TaskState::Zombie as u8 {
                if self.target_pid.is_none() || self.target_pid == Some(child.pid) {
                    let pid = child.pid;
                    let status = child.exit_status.load(Ordering::Acquire);
                    // Don't remove from children list here — do it after poll returns Ready
                    return Poll::Ready((pid, status));
                }
            }
        }

        Poll::Pending
    }
}
```

After `wait4` returns Ready, the caller must remove the zombie child from the children list (reaping). Provide a helper: `Task::reap_child(pid) -> Option<Arc<Task>>` that removes the child Arc from the children vec.
